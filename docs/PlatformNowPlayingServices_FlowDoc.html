NowPlayingService interroga l'endpoint AzuraCast /api/nowplaying/{stationId} ogni 10 secondi tramite PeriodicTimer. Solo se i dati cambiano (confronto tramite IEquatable<NowPlayingInfo>) viene emesso l'evento NowPlayingUpdated.

6 Artwork versioning & race-condition guard

Il caricamento dell'artwork è asincrono e potrebbe completarsi dopo che il brano è cambiato. PlatformNowPlayingStateTracker usa un contatore _artworkVersion per annullare risultati obsoleti.

public PlatformNowPlayingUpdateResult UpdateMetadata(
    string artist, string title,
    string? artworkUrl, int? elapsed,
    int? duration, bool isPlaying)
{
    lock (_sync)           // 🔒 thread-safe
    {
        _artist = artist;
        _title  = title;
        // Artwork cambiato? incrementa version
        if (!string.Equals(_artworkUrl,
              normalizedArtworkUrl, Ordinal))
        {
            _artworkUrl = normalizedArtworkUrl;
            _artworkVersion++;          // 🔢
            artworkVersion = _artworkVersion;
            artworkUrlToLoad = normalizedArtworkUrl;
        }
        return new PlatformNowPlayingUpdateResult(
            CreateSnapshot(),
            artworkUrlToLoad,
            artworkVersion);
    }
}

iOS: Publish su main thread

private static void Publish(
    PlatformNowPlayingSnapshot snapshot,
    MPMediaItemArtwork? artwork)
{
    // ⚠ MPNowPlayingInfoCenter richiede main thread
    MainThread.BeginInvokeOnMainThread(() =>
    {
        if (string.IsNullOrWhiteSpace(snapshot.Title) &&
            string.IsNullOrWhiteSpace(snapshot.Artist))
        {
            MPNowPlayingInfoCenter.DefaultCenter
                .NowPlaying = null!;
            return;
        }
        MPNowPlayingInfo np = new()
        {
            Title        = snapshot.Title,
            Artist       = snapshot.Artist,
            PlaybackRate = snapshot.IsPlaying ? 1.0 : 0.0
        };
        if (artwork is not null) np.Artwork = artwork;
        MPNowPlayingInfoCenter.DefaultCenter.NowPlaying = np;
    });
}

NowPlayingService: polling loop

_pollingTask = Task.Run(async () =>
{
    // Fetch iniziale immediato
    NowPlayingInfo initial = await FetchOnceAsync(station);
    NotifyIfChanged(initial);

    using PeriodicTimer timer =
        new(TimeSpan.FromSeconds(10));
    while (await timer.WaitForNextTickAsync(token))
    {
        if (_isPaused) continue; // pausa = skip
        NowPlayingInfo info = await FetchOnceAsync(station);
        NotifyIfChanged(info);   // emette event solo se ≠
    }
}, token);

AudioService: reconnect guard

// Garantisce un solo reconnect attivo alla volta
private bool TryQueueReconnect()
{
    if (_isShuttingDown) return false;

    // CompareExchange: 0→1 atomico
    if (Interlocked.CompareExchange(
            ref _reconnectGuard, 1, 0) != 0)
        return false; // già in corso

    // Posta sul main thread
    MainThread.BeginInvokeOnMainThread(
        TryReconnectAsync);
    return true;
}
